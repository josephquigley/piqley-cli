How RuleBuilder drives this

// CLI builds context from runtime knowledge let context = RuleEditingContext( availableFields: discoveredFields, // CLI discovers these pluginIdentifier: "my-camera-tags", stages: loadedStages // CLI loads from disk ) // Wizard queries context for options at each step let namespaces = context.availableNamespaces() // → ["original", "read", "exif-tagger"] let fields = context.fields(in: "original") // → [FieldInfo...] // Builder validates each step immediately var builder = RuleBuilder(context: context) builder.setMatch(field: "original:EXIF:LensModel", pattern: "glob:*Sigma*") // → .success builder.addEmit(EmitConfig(field: "keywords", values: ["sigma", "third-party-lens"])) // → .success // Final build + save let rule = try builder.build().get() // CLI writes updated StageConfig to disk
